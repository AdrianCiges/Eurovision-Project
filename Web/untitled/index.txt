<script>
  function sendSong() {

    let song = {
      song: document.getElementById('inputSong').value,
      singer: document.getElementById('inputSinger').value,
      country: document.getElementById('country').value,
    }
    let message = {
      type: "SONG_CREATED",
      value: song
    }
    console.log(song);
    socket.send(JSON.stringify(message))
  }

  function message(event) {
    console.log(event);
    const data = JSON.parse(event.data)
    if (data.type === "COUNTRY_CHOSEN") {
      const selectobject = document.getElementById("country");
      for (let i=0; i<selectobject.length; i++) {
        if (selectobject.options[i].value === data.value) {
          selectobject.remove(i);
        }
      }
    }

  }
</script>
<script>
  let socket = new WebSocket("ws://localhost:8001");
  socket.onmessage = function (event) {
    message(event)
  }
